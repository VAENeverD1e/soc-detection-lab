# PowerShell Suspicious Execution – Detection Rule
# Target SIEM: Microsoft Sentinel (KQL)
# MITRE ATT&CK: T1059.001

SecurityEvent
| where EventID == 4688  // Process creation
| where NewProcessName has "powershell.exe"
    or NewProcessName has "pwsh.exe"
| where CommandLine has_any (
    "-EncodedCommand",
    "-enc ",
    "-ExecutionPolicy Bypass",
    "Invoke-WebRequest",
    "IWR ",
    "Net.WebClient",
    "DownloadString",
    "DownloadFile",
    "IEX(",
    "Invoke-Expression"
)
| project
    TimeGenerated,
    Computer,
    Account,
    ParentProcessName,
    NewProcessName,
    CommandLine
| order by TimeGenerated desc
