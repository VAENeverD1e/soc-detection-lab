# Brute-Force / Password-Spray Login Detection Rule
# Target SIEM: Microsoft Sentinel (KQL)
# MITRE ATT&CK: T1110, T1110.001, T1110.003

// Step 1 – Count failed logins per source IP per account within a 1-minute window
let FailureThreshold = 5;
let TimeWindow = 1m;

SecurityEvent
| where EventID == 4625  // Failed logon
| summarize
    FailureCount = count(),
    DistinctAccounts = dcount(TargetAccount),
    FirstAttempt = min(TimeGenerated),
    LastAttempt = max(TimeGenerated)
    by IpAddress, bin(TimeGenerated, TimeWindow)
| where FailureCount >= FailureThreshold
| join kind=leftouter (
    // Step 2 – Check for a subsequent successful logon from the same IP
    SecurityEvent
    | where EventID == 4624  // Successful logon
    | project SuccessTime = TimeGenerated, IpAddress, TargetAccount
) on IpAddress
| extend SuccessAfterBurst = iff(SuccessTime > LastAttempt, true, false)
| project
    FirstAttempt,
    LastAttempt,
    IpAddress,
    FailureCount,
    DistinctAccounts,
    SuccessAfterBurst
| order by FailureCount desc
